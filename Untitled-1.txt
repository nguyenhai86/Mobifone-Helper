^+g:: {
    data := LoadMobifoneData()
    if !data.HasOwnProp("eligibleCVTN") {
        MsgBox "Error: eligibleCVTN not found in data"
        return
    }

    ; Get eligible packages as array
    eligibleCVTN := data["eligibleCVTN"]
    if !IsObject(eligibleCVTN) {
        MsgBox "Error: eligibleCVTN is not in correct format"
        return
    }

    packageName := GetSelectedText()
    if !RegExMatch(packageName, "^[a-zA-Z0-9]+$") {
        MsgBox Format("Package '{1}' is invalid", packageName)
        return
    }

    ; Check if package is in eligible list
    isCVTN := false
    for pkg in eligibleCVTN {
        if (pkg = packageName) {
            isCVTN := true
            break
        }
    }

    ; Show results
    guiTitle := "CVTN Check"
    gui := Gui(, guiTitle)
    gui.Add("Text", "x10 y20", Format("Package: {1}", packageName))
    gui.Add("Text", "x10 y50 " (isCVTN ? "cGreen" : "cRed"),
        isCVTN ? "✓ CVTN eligible" : "✗ Not eligible for CVTN")
    gui.OnEvent("Escape", (*) => gui.Destroy())
    gui.Show()
}
